# 概览
> gum是基于php的开箱即用代码库。


# 数据库
```php
// 链接数据库
$db = new db("table","password","root","127.0.0.1",3306);

// 查询一条记录
$row=$db->row("SELECT * FROM table");

// 查询记录集合
$rows=$db->rows("SELECT * FROM table");

// 查询记录总数
$count=$db->count("SELECT * FROM table");

// 插入
$db->insert("table",[
	"title" => "title",
	"content" => "content",
]);

// 获取新插入的自动编号id
$db->id();

// 更新
$db->update("table",[
	"title" => "title",
	"content" => "content",
],"id=1");

// 删除
$db->delete("table","id=1");

// 清空表数据
$db->clear(["table1","table2"]);

// 备份数据库导出sql语句
$db->export();

```

	"subject"  => "test",
	"body"     => "test",
